//3.在屏幕上打印杨辉三角。
#include <iostream>
#include <stdio.h>
using namespace std;

int main()
{
	int arr[10][10] = { { 1 }, {1,1} };

	for (int i = 2; i < 10; ++i)
	{
		int j = 0;
		arr[i][j] = 1;
		for (j = 1; j < i-1; ++j)
		{
			arr[i][j] = arr[i - 1][j] + arr[i - 1][j - 1];
		}
		arr[i][j] = 1;
	}

	for (int i = 0; i < 10; ++i)
	{
		for (int a = 0; a < (9 - i); ++a)
			cout << " ";
		for (int j = 0; j < i; ++j)
			printf(" %2d", arr[i][j]);
		cout << endl;
	}
	return 0;
}
